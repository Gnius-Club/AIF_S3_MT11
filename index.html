# Importaciones necesarias

from network import WLAN, STA_IF

from machine import Pin, SPI # SPI para RC522

# --- Importar la librer√≠a espec√≠fica del sensor RC522 RFID ---
# ¬°¬°NOTA IMPORTANTE: Deber√°n tener el archivo 'mfrc522.py' en la ra√≠z de su Micro:bit!!
# Este archivo es la librer√≠a que Xolo asume para interactuar con el RC522 via SPI.
from mfrc522 import MFRC522

# --- Configuraci√≥n de la red Wi-Fi ---
SSID = '[SSID_WIFI_CLASE]'
PASSWORD = '[PASSWORD_WIFI_CLASE]'

# --- Configuraci√≥n de la API de SmartIOT.space (ThingSpeak-compatible) ---
API_KEY = 'YOUR_THINGSPEAK_WRITE_API_KEY'
CHANNEL_ID = 'YOUR_THINGSPEAK_CHANNEL_ID'

# Definir pin para LED de confirmaci√≥n

# --- Configuraci√≥n SPI para RC522 (adaptar a pines de IOTbit si es diferente) ---
# Pines SPI en Micro:bit: MISO=P16, MOSI=P15, SCK=P14. RST y CS deben definirse.
# Asumo que IOTbit tiene mapeo o se usan pines libres para RST y CS.
# Generalmente: MISO=P16, MOSI=P15, SCK=P14, RST=P13, CS=P4 (ejemplo com√∫n)
spi = SPI(1, baudrate=1000000, polarity=0, phase=0, bits=8, firstbit=SPI.MSB,
            sck=Pin(14), mosi=Pin(15), miso=Pin(16))
# Definir pines CS (Chip Select) y RST (Reset) para el RC522
sck = Pin(14)
mosi = Pin(15)
miso = Pin(16)
rst = Pin(13) # Pin de reset del RC522
cs = Pin(4)  # Pin Chip Select del RC522
rfid = MFRC522(spi, cs, rst)

# --- Variable de estado para gestionar el env√≠o de IDs √∫nicos (solo una vez por detecci√≥n) ---
uid_leido_anteriormente = None # Almacena el √∫ltimo UID le√≠do para evitar re-env√≠os

# --- Funci√≥n para conectar a Wi-Fi (reutilizada) ---
def connect_wifi():
    wlan = WLAN(STA_IF)
    wlan.active(True)
    if not wlan.isconnected():
        print('Intentando conectar a Wi-Fi...')
        display.scroll("WIFI...")
        wlan.connect(SSID, PASSWORD)
        timeout = 0
        while not wlan.isconnected() and timeout < 30:
            display.show(Image.ASLEEP)
            time.sleep_ms(300)
            display.clear()
            time.sleep_ms(300)
            timeout += 1
    if wlan.isconnected():
        print('Conexi√≥n Wi-Fi exitosa:', wlan.ifconfig())
        display.scroll("CONECTADO")
        return True
    else:
        print('Fallo la conexi√≥n Wi-Fi.')
        display.scroll("WIFI ERROR")
        return False

# --- Funci√≥n para enviar un dato a ThingSpeak (reutilizada para UID) ---
def send_uid_to_thingspeak(uid_hex_str, field_num=1):
    try:
        # ThingSpeak puede tener problemas con caracteres especiales. urequests.quote codifica el string.
        url_with_params = THINGSPEAK_URL + '?api_key=' + API_KEY + '&field' + str(field_num) + '=' + urequests.quote(uid_hex_str)

        print(f"Enviando UID: {uid_hex_str}")
        display.scroll("UID...")
        response = urequests.get(url_with_params)

        if response.status_code == 200:
            print("UID enviado exitosamente.")
            display.show(Image.CHECK)
            time.sleep_ms(300)
            display.clear()
            return True
        else:
            print(f"Error UID. C√≥digo: {response.status_code} Mensaje: {response.text}")
            display.show(Image.X)
            time.sleep(1)
            display.clear()
            return False
    except Exception as e:
        print(f"Excepci√≥n al enviar UID: {e}")
        display.show(Image.SAD)
        time.sleep(1)
        display.clear()
        return False

# --- Bucle principal del programa ---
if connect_wifi():
    display.scroll("Lector ON!")
    time.sleep(1)

    while True:
        display.show(Image.ASLEEP) # Icono "Esperando Tag"
        led_verde_confirmacion.write_digital(0) # Apagar LED

        (status, tag_type) = rfid.request(rfid.REQIDL) # Pide si hay etiqueta cerca

        if status == rfid.OK:
            (status, uid_raw) = rfid.SelectTag(uid_raw) # Lee el UID

            if status == rfid.OK:
                uid_hex = ''.join(['%02x' % i for i in uid_raw]).upper() # Convierte UID a formato hexadecimal
                print(f"Tag detectado - UID: {uid_hex}")

                if uid_hex != uid_leido_anteriormente: # Si es un nuevo tag o el mismo pero ya se retir√≥
                    led_verde_confirmacion.write_digital(1) # Encender LED
                    display.show(Image.YES) # Icono de confirmaci√≥n
                    send_uid_to_thingspeak(uid_hex, 1) # Enviar UID a ThingSpeak (field1)
                    uid_leido_anteriormente = uid_hex # Actualizar el √∫ltimo UID le√≠do
                    time.sleep(1) # Pausa para evitar lecturas dobles en el mismo ciclo
                else:
                    print("Mismo Tag, no reenviado.")
                    display.show(Image.HAPPY) # Icono de ya le√≠do

            else:
                print("Error al leer Tag.")
                display.show(Image.NO)
        else:
            # Si no hay tag detectado, y antes hab√≠a uno, resetear la bandera
            if uid_leido_anteriormente is not None:
                print("Tag retirado. Reseteando estado.")
                uid_leido_anteriormente = None # Resetear la bandera
                # Opcional: enviar un mensaje a ThingSpeak de "Tag retirado"
                # send_uid_to_thingspeak("Tag Retirado", 1)

        time.sleep_ms(100) # Peque√±a pausa para evitar sobrecargar el loop
else:
    print("El sistema se detuvo debido a un error de conexi√≥n Wi-Fi. Reinicie para intentar de nuevo.")
    display.scroll("ERROR Fatal")
